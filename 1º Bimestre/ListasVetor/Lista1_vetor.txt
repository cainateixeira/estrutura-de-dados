1) Crie um classe com um vetor de inteiros de 10 posições e solicite ao usuário para
inserir os elementos do vetor, mostre os elementos no final;

import java.util.Scanner;

public class VetorInteiros {
    public static void main(String[] args) {
        int[] vetor = new int[10];
        Scanner sc = new Scanner(System.in);
        
        for (int i = 0; i < vetor.length; i++) {
            System.out.print("Insira o elemento " + (i+1) + ": ");
            vetor[i] = sc.nextInt();
        }
        
        System.out.println("\nElementos do vetor:");
        for (int i = 0; i < vetor.length; i++) {
            System.out.println(vetor[i]);
        }
    }
}

2) Crie uma classe que tenha um vetor de inteiros de 20 posições, e solicite ao usuário informar
um valor,  depois mostre o valor informado e sua posição;

import java.util.Scanner;

public class VetorPosicao {
    public static void main(String[] args) {
        int[] vetor = new int[20];
        Scanner sc = new Scanner(System.in);
        
        for (int i = 0; i < vetor.length; i++) {
            System.out.print("Insira o elemento " + (i+1) + ": ");
            vetor[i] = sc.nextInt();
        }
        
        System.out.print("Informe um valor: ");
        int valor = sc.nextInt();
        
        int posicao = -1;
        for (int i = 0; i < vetor.length; i++) {
            if (vetor[i] == valor) {
                posicao = i;
                break;
            }
        }
        
        if (posicao != -1) {
            System.out.println("O valor " + valor + " foi encontrado na posição " + posicao);
        } else {
            System.out.println("O valor " + valor + " não foi encontrado no vetor");
        }
    }
}

3) Crie uma classe que tenha um vetor de inteiros de 30 posições e mostre a soma total dos elementos;

public class VetorSoma {
    public static void main(String[] args) {
        int[] vetor = new int[30];
        
        for (int i = 0; i < vetor.length; i++) {
            vetor[i] = i+1;
        }
        
        int soma = 0;
        for (int i = 0; i < vetor.length; i++) {
            soma += vetor[i];
        }
        
        System.out.println("A soma total dos elementos do vetor é: " + soma);
    }
}

7)  Crie uma aplicação para informar 10 números , remova os números ímpares da lista, e no final mostre
a lista sem os valores ímpares.

import java.util.ArrayList;
import java.util.Scanner;

public class RemoverNumerosImpares {
    public static void main(String[] args) {
        ArrayList<Integer> numeros = new ArrayList<>();
        Scanner sc = new Scanner(System.in);
        
        for (int i = 0; i < 10; i++) {
            System.out.print("Insira o número " + (i+1) + ": ");
            numeros.add(sc.nextInt());
        }
        
        for (int i = 0; i < numeros.size(); i++) {
            if (numeros.get(i) % 2 != 0) {
                numeros.remove(i);
                i--;
            }
        }
        
        System.out.println("A lista sem os números ímpares é:");
        for (int i = 0; i < numeros.size(); i++) {
            System.out.println(numeros.get(i));
        }
    }
}

9) Crie uma aplicação que controle as notas de um aluno, para isso deve armazenar, o nome, a matricula,
disciplina e as 4 notas. Mostre os dados informados, as notas de cada bimestre a sua média e informar se
o aluno está aprovado ou reprovado sendo que a média para aprovação é 7.

import java.util.Scanner;

public class Aluno {
    private String nome;
    private String matricula;
    private String disciplina;
    private double[] notas = new double[4];

    public Aluno(String nome, String matricula, String disciplina, double[] notas) {
        this.nome = nome;
        this.matricula = matricula;
        this.disciplina = disciplina;
        this.notas = notas;
    }

    public void exibirDados() {
        System.out.println("Nome: " + nome);
        System.out.println("Matrícula: " + matricula);
        System.out.println("Disciplina: " + disciplina);
        System.out.println("Notas:");
        for (int i = 0; i < notas.length; i++) {
            System.out.println("  Bimestre " + (i + 1) + ": " + notas[i]);
        }
        System.out.println("Média: " + calcularMedia());
        System.out.println("Situação: " + (calcularMedia() >= 7 ? "Aprovado" : "Reprovado"));
    }

    public double calcularMedia() {
        double soma = 0;
        for (int i = 0; i < notas.length; i++) {
            soma += notas[i];
        }
        return soma / notas.length;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Digite o nome do aluno: ");
        String nome = sc.nextLine();

        System.out.print("Digite a matrícula do aluno: ");
        String matricula = sc.nextLine();

        System.out.print("Digite a disciplina: ");
        String disciplina = sc.nextLine();

        double[] notas = new double[4];
        for (int i = 0; i < notas.length; i++) {
            System.out.print("Digite a nota do " + (i + 1) + "º bimestre: ");
            notas[i] = sc.nextDouble();
        }

        Aluno aluno = new Aluno(nome, matricula, disciplina, notas);
        aluno.exibirDados();
    }
}
