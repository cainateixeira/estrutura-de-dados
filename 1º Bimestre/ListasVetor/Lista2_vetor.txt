1) Crie uma aplicação onde o usuário tenha 3 opções:
1 – Inserir um nome;
2 -  remover o nome da posição informada;
3 - exibir todos os nomes da lista.

import java.util.ArrayList;
import java.util.Scanner;

public class ListaNomes {

    public static void main(String[] args) {
        ArrayList<String> nomes = new ArrayList<String>();
        Scanner scanner = new Scanner(System.in);
        int opcao;

        do {
            System.out.println("Escolha uma opção:");
            System.out.println("1 - Inserir um nome");
            System.out.println("2 - Remover um nome");
            System.out.println("3 - Exibir todos os nomes");
            System.out.println("0 - Sair");
            opcao = scanner.nextInt();

            switch (opcao) {
                case 1:
                    System.out.println("Digite o nome:");
                    String nome = scanner.next();
                    nomes.add(nome);
                    System.out.println("Nome adicionado com sucesso!");
                    break;
                case 2:
                    System.out.println("Digite a posição do nome que deseja remover:");
                    int posicao = scanner.nextInt();
                    if (posicao >= 0 && posicao < nomes.size()) {
                        nomes.remove(posicao);
                        System.out.println("Nome removido com sucesso!");
                    } else {
                        System.out.println("Posição inválida!");
                    }
                    break;
                case 3:
                    System.out.println("Lista de nomes:");
                    for (int i = 0; i < nomes.size(); i++) {
                        System.out.println(i + " - " + nomes.get(i));
                    }
                    break;
                case 0:
                    System.out.println("Saindo...");
                    break;
                default:
                    System.out.println("Opção inválida!");
                    break;
            }

            System.out.println();

        } while (opcao != 0);

        scanner.close();
    }

}

2) Um vendedor precisa controlar suas vendas e comissões, a cada mês ele informa o total de vendas do mês, porém ele precisa saber qual o total de comissão que ele recebeu durante todo o ano. Para calcular a comissão deve-se obedecer os seguintes critérios:
- Valor total até R$500,00 comissão 0,5%;

- Valor total entre R$501,00 até R$800,00 comissão 0,8%;

- Valor total entre R$801,00 até R$1000,00 comissão 1%;

- Valor total entre R$1001,00 até R$3000,00 comissão 1,5%;

- Valor acima de R$3000,00 comissão 2%;

Mostre no final o valor total da venda de cada mês, a porcentagem de comissão e o valor da comissão de seus respectivos meses.

import java.util.Scanner;

public class VendasComissao {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        double valorTotalAno = 0;
        double comissaoAno = 0;

        for (int mes = 1; mes <= 12; mes++) {
            System.out.println("Digite o valor total de vendas do mês " + mes + ":");
            double valorTotalMes = scanner.nextDouble();
            valorTotalAno += valorTotalMes;
            double comissaoMes = 0;

            if (valorTotalMes <= 500) {
                comissaoMes = valorTotalMes * 0.005;
            } else if (valorTotalMes <= 800) {
                comissaoMes = valorTotalMes * 0.008;
            } else if (valorTotalMes <= 1000) {
                comissaoMes = valorTotalMes * 0.01;
            } else if (valorTotalMes <= 3000) {
                comissaoMes = valorTotalMes * 0.015;
            } else {
                comissaoMes = valorTotalMes * 0.02;
            }

            comissaoAno += comissaoMes;

            System.out.printf("Valor total de vendas do mês %d: R$ %.2f\n", mes, valorTotalMes);
            System.out.printf("Porcentagem de comissão do mês %d: %.2f%%\n", mes, comissaoMes * 100 / valorTotalMes);
            System.out.printf("Valor de comissão do mês %d: R$ %.2f\n", mes, comissaoMes);
            System.out.println();
        }

        System.out.printf("Valor total de vendas do ano: R$ %.2f\n", valorTotalAno);
        System.out.printf("Valor total de comissão do ano: R$ %.2f\n", comissaoAno);

        scanner.close();
    }
}
